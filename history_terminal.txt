    id INT PRIMARY KEY AUTO_INCREMENT,
    Name VARCHAR(50),
    Commands VARCHAR(255),
    Type VARCHAR(50),
    Birthday DATE

CREATE TABLE Horses_and_Donkeys AS

CREATE TABLE Young_animals AS
SELECT 
    id,
    Name,
    Commands,
    Type,
    Birthday,
    TIMESTAMPDIFF(MONTH, Birthday, CURDATE()) AS Age_by_month
FROM Animals
WHERE TIMESTAMPDIFF(YEAR, Birthday, CURDATE()) BETWEEN 1 AND 3;

SELECT * FROM Young_animals;

CREATE TABLE All_animals_combined AS
SELECT *, 'Animals' AS Source FROM Animals
UNION ALL
SELECT *, 'Young_animals' AS Source FROM Young_animals
UNION ALL
SELECT *, 'Horses_and_Donkeys' AS Source FROM Horses_and_Donkeys;

SELECT * FROM All_animals_combined;
